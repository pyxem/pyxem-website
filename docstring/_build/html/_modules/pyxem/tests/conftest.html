            [[1, 2, 3], [0, 2, 0]],
            [[1, 2, 3], [1, 0, 0]],

    }
    lattice = diffpy.structure.Lattice(1, 1, 1, 90, 90, 90)

        diffpy.structure.Structure(lattice=lattice)
    ]
    return library
